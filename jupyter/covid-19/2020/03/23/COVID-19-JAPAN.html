import pandas as pd
import altair as alt
import requests
import io

URL = "https://raw.githubusercontent.com/kaz-ogiwara/covid19/master/data/individuals.csv"

r = requests.get(URL)
df = pd.read_csv(io.BytesIO(r.content), sep=",")
df = df.drop(0).replace('代', '', regex=True)
df['date'] = df['確定年'].astype(str) + '-' + df['確定月'].astype(str) + '-' + df['確定日'].astype(str)
