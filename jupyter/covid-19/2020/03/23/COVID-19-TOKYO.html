import pandas as pd
import altair as alt
import requests
import io

URL = "https://stopcovid19.metro.tokyo.lg.jp/data/130001_tokyo_covid19_patients.csv"

r = requests.get(URL)
df = pd.read_csv(io.BytesIO(r.content), sep=",")
df = df.drop(0).replace('代', '', regex=True)
